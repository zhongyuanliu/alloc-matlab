390function [A,N] = N_sided_linear(error)
391%%
392N = floor(pi / (acos(1 - error))) + 1;
393%     r = - R * cos(pi / N);%change <= to >=
394phi = zeros(N,1);
395A = zeros(N,2);
396for i = 0 : (N - 1)
397    phi(i + 1) = (2 * i + 1) * pi / N;
398    A(i + 1,1) = - cos(phi(i + 1));%change <= to >=
399    A(i + 1,2) = - sin(phi(i + 1));%change <= to >=
400end

402end
