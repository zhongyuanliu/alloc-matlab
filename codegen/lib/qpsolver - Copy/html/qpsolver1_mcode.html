1function [solution,alloc_out,status]=qpsolver(thruster_data,T_r,N_enabled_thruster,rudder_table0,no_azi_angle_constr,method)
  2% method == 1 qpkwik
  3% method == 2 primal active set
  4coder.inline('never');
  5rudder_table = rudder_table0';%!!!!!!!!!!!!!!!
  6% no_azi_angle_constr = 1;
  7%% rudder table
  8persistent Rangle_T Rangle_Fangle Fangle_Rangle rudder_dat_init;
  9if isempty(rudder_dat_init)%assume this is only one type of rudder
 10    rudder_dat_init = false;
 11    N_poly = no_azi_angle_constr + 2;
 12    [Rangle_T, ~] = polyfit(rudder_table(:,1), rudder_table(:,5),N_poly);%rudder angle vs T
 13    [Rangle_Fangle, ~] = polyfit(rudder_table(:,1), rudder_table(:,4),N_poly);%rudder angle vs F angle
 14    [Fangle_Rangle, ~] = polyfit(rudder_table(:,4), rudder_table(:,1),N_poly);%F angle vs rudder angle
 15end
 16NA_fpp = size(rudder_table,1) * 2;
 17% coder.varsize('alloc_out');%!!!!!!!!!very important???
 18alloc_out = struct ('label',{0,0,0,0,0,0,0,0}, ...
 19    'enable',1, ...
 20    'TSP',0, ...
 21    'ASP',0, ...
 22    'Tx',0, ...
 23    'Ty',0, ...
 24    'T',0, ...
 25    'phi',0, ...
 26    'Tm',0);
 27flag_run = 1;
 28count_run = 0;
 29mark8 = zeros(8,1);
 30solution = zeros(N_enabled_thruster * 2 + 3,1);   
 31while flag_run
 32    flag_run = 0;
 33    count_run = count_run + 1;
 34    [thruster_data,Nvar,H,f,A,b,Aeq,beq,x0] = pre_qp(thruster_data,T_r,N_enabled_thruster,NA_fpp,rudder_table,Rangle_T,Rangle_Fangle,no_azi_angle_constr,method);
 35    solution = x0;
 36    status = 0;
 37    
 38    % phi_f=atan2(rudder_table(:,3),rudder_table(:,2));%angle of Fx Fy
 39    % rudder_table(:,end+1)=phi_f;
 40    
 41    MaxIter=100;
 42    FeasibilityTol=1e-8;
 43    
 44    [L,p] = Chol_fc(H);
 45    % before chol
 46    
 47    if p==0
 48        switch method
 49            case 1
 50                Linv = inv(L);
 51                opt = mpcqpsolverOptions;
 52                opt.MaxIter = MaxIter;
 53                opt.IntegrityChecks = false;
 54                opt.FeasibilityTol = FeasibilityTol;
 55                iA0 = false(size(b));
 56                %     if isempty(iA)
 57                %         iA = iA0;
 58                %     end
 59                
 60                [solution,status,lambda] = Mpcqpsolver(Linv,f,A,b,Aeq,beq,iA0,opt);%faster to use iA instead of iA0
 61                %     [solution1,fval,exitflag] = quadprog(H,f,A,b,Aeq,beq)
 62            case 2
 63                %     x0=[x0;0;0;0];
 64                col_Aeq=size(Aeq,2);
 65                %     ????????????????ux,uy,?????????s1?s2?s3???
 66                x0(col_Aeq - 2 :end) = Aeq(1:3,col_Aeq - 2 :end)\(beq(1:3) - Aeq(1:3,1:col_Aeq - 3) * x0(1:col_Aeq - 3));%does not work if there is tunnel!!!!!!!!!
 67                
 68                [solution,status] = QPACM(H,-f,Aeq,beq,-A,-b,x0);
 69            otherwise
 70                 solution = zeros(N_enabled_thruster * 2 + 3,1);   
 71                % x = quadprog(H,f,A,b,Aeq,beq);
 72        end
 73    else
 74        solution = zeros(Nvar,1);
 75        status = -2;
 76    end
 77    
 78    solution = solution.*(abs(solution)>=FeasibilityTol);%!!!!!avoid nuemrical problem
 79    
 80    % output
 81    numOfTunnel = 0;
 82    numOfAzi = 0;
 83    numOfFpp = 0;
 84    count_target_thr = 0;
 85    for i=1:N_enabled_thruster
 86        alloc_out(i).label = thruster_data(i).label;
 87        alloc_out(i).enable = thruster_data(i).enable;
 88        
 89        switch thruster_data(i).type
 90            case 4
 91                alloc_out(i).Tx = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1);
 92                alloc_out(i).Ty = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2);
 93                alloc_out(i).T = alloc_out(i).Ty;
 94                alloc_out(i).phi = atan2(solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2), ...
 95                    solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1));
 96                alloc_out(i).phi = thruster_data(i).phi * (abs([alloc_out(i).T]) <= FeasibilityTol) ...
 97                    + alloc_out(i).phi .* (abs(alloc_out(i).T) > FeasibilityTol);
 98                alloc_out(i).Tm = alloc_out(i).Ty * (thruster_data(i).x ...
 99                    - thruster_data(i). x0) + alloc_out(i).Tx ...
100                    * (thruster_data(i).y0 - thruster_data(i).y);
101                
102                numOfTunnel = numOfTunnel + 1;
103            case 3
104                alloc_out(i).Tx = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1);
105                alloc_out(i).Ty = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2);
106                alloc_out(i).T = sqrt(alloc_out(i).Tx^2 + alloc_out(i).Ty^2);
107                alloc_out(i).phi = atan2(solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2), ...
108                    solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1));
109                alloc_out(i).phi = thruster_data(i).phi * (abs([alloc_out(i).T]) <= FeasibilityTol) ...
110                    + alloc_out(i).phi .* (abs(alloc_out(i).T) > FeasibilityTol);
111                alloc_out(i).Tm = alloc_out(i).Ty * (thruster_data(i).x ...
112                    - thruster_data(i). x0) + alloc_out(i).Tx ...
113                    * (thruster_data(i).y0 - thruster_data(i).y);
114                if abs(thruster_data(i).phi_max - thruster_data(i).phi_min)> FeasibilityTol%means the thruster has physical constraint
115                    if angle_dist(thruster_data(i).phi_min,alloc_out(i).phi) ...
116                            > angle_dist(thruster_data(i).phi_min,thruster_data(i).phi_max) ...
117                            + FeasibilityTol                  % if solution is inside fb zone
118                        
119                        thruster_data(i).constr_angle = 1;
120                        count_target_thr = count_target_thr + 1;
121                        mark8(i) = 1;%when mark8(i)==1, the thruster will be constrained from now on in while loop
122                        flag_run = 1;
123                    end
124                    
125                    if (abs(alloc_out(i).phi - thruster_data(i).phi_min) <= FeasibilityTol ...
126                            || abs(alloc_out(i).phi - thruster_data(i).phi_max) <= FeasibilityTol) ...
127                            && abs(alloc_out(i).T - thruster_data(i).Tmin) <= FeasibilityTol ...
128                            &&  mark8(i) == 0
129                        thruster_data(i).constr_angle = 0;
130                        flag_run = 1;% remove constr and run again
131                    end
132                end
133                numOfAzi = numOfAzi + 1;
134            case 2
135                alloc_out(i).Tx = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1);
136                alloc_out(i).Ty = solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2);
137                alloc_out(i).T = sqrt(alloc_out(i).Tx^2 + alloc_out(i).Ty^2);
138                alloc_out(i).phi = atan2(solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 2), ...
139                    solution(numOfTunnel * 2 + numOfAzi * 2 + numOfFpp * 2 + 1));
140                alloc_out(i).phi = thruster_data(i).phi * (abs([alloc_out(i).T]) <= FeasibilityTol) ...
141                    + alloc_out(i).phi .* (abs(alloc_out(i).T) > FeasibilityTol);
142                alloc_out(i).phi = sign(alloc_out(i).phi) * polyval(Fangle_Rangle,abs(alloc_out(i).phi));
143                alloc_out(i).Tm = alloc_out(i).Ty * (thruster_data(i).x ...
144                    - thruster_data(i). x0) + alloc_out(i).Tx ...
145                    * (thruster_data(i).y0 - thruster_data(i).y);
146                numOfFpp = numOfFpp + 1;
147        end
148    end
149end
150
151% status
152% chck_sol = [[thruster_data(:).phi_min]' [thruster_data(:).phi_]' [alloc_out.phi]' [thruster_data.phiPlus]' [thruster_data(:).phi_max]'];
153% check_angle(chck_sol(:,1),chck_sol(:,2),chck_sol(:,3),chck_sol(:,4),chck_sol(:,5))
154% solution
155end
