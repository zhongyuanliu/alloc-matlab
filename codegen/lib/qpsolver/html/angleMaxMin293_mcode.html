 3function [phi_,phiPlus]=angleMaxMin(angle_start,angle_end,phi,phi1,phi2)
 4%% up x axis, right y axis, angle starts from x axis, -pi to pi, clockwise
 5% phi: current angle
 6% phi1 to phi2
 7%!!!!!!!!!!!!!!!!!!!!!watch out for FeasibilityTol
 8FeasibilityTol = 1e-8;
 9angle_start = Pi_toPi(angle_start);
10angle_end = Pi_toPi(angle_end);
11phi = Pi_toPi(phi);
12phi1 = Pi_toPi(phi1);
13phi2 = Pi_toPi(phi2);
14effective_dist = angle_dist(angle_start,angle_end);
15start_phi1 = angle_dist(angle_start,phi1);
16phi1_end = angle_dist(phi1,angle_end);
17phi1_phi2 = angle_dist(phi1,phi2);
18phi1_start = angle_dist(phi1,angle_start);
19phi1_phi = angle_dist(phi1,phi);
20start_phi = angle_dist(angle_start,phi);
21start_phi2 = angle_dist(angle_start,phi2);
22phi_ = 0;
23phiPlus = 0;
24if abs(angle_start - angle_end) < FeasibilityTol%angle_start == angle_end means no forbidden zone
25    phi_ = phi1;
26    phiPlus = phi2;
27else
28    
29if effective_dist >= start_phi1 %phi1 is at the range from start to end
30    if phi1_end >= phi1_phi2 %phi2 is at the range from phi1 to end
31        phi_ = phi1;
32        phiPlus = phi2;
33    elseif phi1_phi2 > phi1_end ...
34            && phi1_phi2 <= phi1_start
35        phi_ = phi1;
36        phiPlus = angle_end;
37    else %phi2 is at the range from start to phi1
38        if phi1_phi <= phi1_end%phi is at the range from phi1 to end
39            phi_ = phi1;
40            phiPlus = angle_end;
41        elseif start_phi <= start_phi2%phi is at the range from start to phi2
42            phi_ = angle_start;
43            phiPlus = phi2;
44        end
45        
46    end
47else%phi1 is at the range from end to start
48    phi_ = angle_start;
49    phiPlus = phi2;
50end

52end
